 40 std::vector<int> convertJIntArrayToVector(JNIEnv* env, jintArray inputs) {
 41   int size = static_cast<int>(env->GetArrayLength(inputs));
 42   std::vector<int> outputs(size, 0);
 43   jint* ptr = env->GetIntArrayElements(inputs, nullptr);
 44   if (ptr == nullptr) {
 45     throwException(env, kIllegalArgumentException,
 46                    "Array has empty dimensions.");
 47     return {};
 48   }
 49   for (int i = 0; i < size; ++i) {
 50     outputs[i] = ptr[i];
 51   }
 52   env->ReleaseIntArrayElements(inputs, ptr, JNI_ABORT);
 53   return outputs;
 54 }

 56 int getDataType(TfLiteType data_type) {
 57   switch (data_type) {
 58     case kTfLiteFloat32:
 59       return 1;
 60     case kTfLiteInt32:
 61       return 2;
 62     case kTfLiteUInt8:
 63       return 3;
 64     case kTfLiteInt64:
 65       return 4;
 66     case kTfLiteString:
 67       return 5;
 68     default:
 69       return -1;
 70   }
 71 }

 73 void printDims(char* buffer, int max_size, int* dims, int num_dims) {
 74   if (max_size <= 0) return;
 75   buffer[0] = '?';
 76   int size = 1;
 77   for (int i = 1; i < num_dims; ++i) {
 78     if (max_size > size) {
 79       int written_size =
 80           snprintf(buffer + size, max_size - size, ",%d", dims[i]);
 81       if (written_size < 0) return;
 82       size += written_size;
 83     }
 84   }
 85 }

 87 bool areDimsDifferent(JNIEnv* env, TfLiteTensor* tensor, jintArray dims) {
 88   int num_dims = static_cast<int>(env->GetArrayLength(dims));
 89   jint* ptr = env->GetIntArrayElements(dims, nullptr);
 90   if (ptr == nullptr) {
 91     throwException(env, kIllegalArgumentException,
 92                    "Empty dimensions of input array.");
 93     return true;
 94   }
 95   if (tensor->dims->size != num_dims) {
 96     return true;
 97   }
 98   for (int i = 0; i < num_dims; ++i) {
 99     if (ptr[i] != tensor->dims->data[i]) {
100       return true;
101     }
102   }
103   env->ReleaseIntArrayElements(dims, ptr, JNI_ABORT);
104   return false;
105 }

115 JNIEXPORT jobjectArray JNICALL
116 Java_org_tensorflow_lite_NativeInterpreterWrapper_getInputNames(JNIEnv* env,
117                                                                 jclass clazz,
118                                                                 jlong handle) {
119   tflite::Interpreter* interpreter = convertLongToInterpreter(env, handle);
120   if (interpreter == nullptr) return nullptr;
121   jclass string_class = env->FindClass("java/lang/String");
122   if (string_class == nullptr) {
123     throwException(env, kUnsupportedOperationException,
124                    "Internal error: Can not find java/lang/String class to get "
125                    "input names.");
126     return nullptr;
127   }
128   size_t size = interpreter->inputs().size();
129   jobjectArray names = static_cast<jobjectArray>(
130       env->NewObjectArray(size, string_class, env->NewStringUTF("")));
131   for (int i = 0; i < size; ++i) {
132     env->SetObjectArrayElement(names, i,
133                                env->NewStringUTF(interpreter->GetInputName(i)));
134   }
135   return names;
136 }

257 // Verifies whether the model is a flatbuffer file.
258 class JNIFlatBufferVerifier : public tflite::TfLiteVerifier {
259  public:
260   bool Verify(const char* data, int length,
261               tflite::ErrorReporter* reporter) override {
262     if (!VerifyModel(data, length)) {
263       reporter->Report("The model is not a valid Flatbuffer file");
264       return false;
265     }
266     return true;
267   }
268 };

270 JNIEXPORT jlong JNICALL
271 Java_org_tensorflow_lite_NativeInterpreterWrapper_createModel(
272     JNIEnv* env, jclass clazz, jstring model_file, jlong error_handle) {
273   BufferErrorReporter* error_reporter =
274       convertLongToErrorReporter(env, error_handle);
275   if (error_reporter == nullptr) return 0;
276   const char* path = env->GetStringUTFChars(model_file, nullptr);
277 
278   std::unique_ptr<tflite::TfLiteVerifier> verifier;
279   verifier.reset(new JNIFlatBufferVerifier());
280 
281   auto model = tflite::FlatBufferModel::VerifyAndBuildFromFile(
282       path, verifier.get(), error_reporter);
283   if (!model) {
284     throwException(env, kIllegalArgumentException,
285                    "Contents of %s does not encode a valid "
286                    "TensorFlowLite model: %s",
287                    path, error_reporter->CachedErrorMessage());
288     env->ReleaseStringUTFChars(model_file, path);
289     return 0;
290   }
291   env->ReleaseStringUTFChars(model_file, path);
292   return reinterpret_cast<jlong>(model.release());
293 }

295 JNIEXPORT jlong JNICALL
296 Java_org_tensorflow_lite_NativeInterpreterWrapper_createModelWithBuffer(
297     JNIEnv* env, jclass /*clazz*/, jobject model_buffer, jlong error_handle) {
298   BufferErrorReporter* error_reporter =
299       convertLongToErrorReporter(env, error_handle);
300   if (error_reporter == nullptr) return 0;
301   const char* buf =
302       static_cast<char*>(env->GetDirectBufferAddress(model_buffer));
303   jlong capacity = env->GetDirectBufferCapacity(model_buffer);
304   if (!VerifyModel(buf, capacity)) {
305     throwException(env, kIllegalArgumentException,
306                    "ByteBuffer is not a valid flatbuffer model");
307     return 0;
308   }
309 
310   auto model = tflite::FlatBufferModel::BuildFromBuffer(
311       buf, static_cast<size_t>(capacity), error_reporter);
312   if (!model) {
313     throwException(env, kIllegalArgumentException,
314                    "ByteBuffer does not encode a valid model: %s",
315                    error_reporter->CachedErrorMessage());
316     return 0;
317   }
318   return reinterpret_cast<jlong>(model.release());
319 }

321 JNIEXPORT jlong JNICALL
322 Java_org_tensorflow_lite_NativeInterpreterWrapper_createInterpreter(
323     JNIEnv* env, jclass clazz, jlong model_handle, jlong error_handle,
324     jint num_threads) {
325   tflite::FlatBufferModel* model = convertLongToModel(env, model_handle);
326   if (model == nullptr) return 0;
327   BufferErrorReporter* error_reporter =
328       convertLongToErrorReporter(env, error_handle);
329   if (error_reporter == nullptr) return 0;
330   auto resolver = ::tflite::CreateOpResolver();
331   std::unique_ptr<tflite::Interpreter> interpreter;
332   TfLiteStatus status = tflite::InterpreterBuilder(*model, *(resolver.get()))(
333       &interpreter, static_cast<int>(num_threads));
334   if (status != kTfLiteOk) {
335     throwException(env, kIllegalArgumentException,
336                    "Internal error: Cannot create interpreter: %s",
337                    error_reporter->CachedErrorMessage());
338     return 0;
339   }
340   // Note that tensor allocation is performed explicitly by the owning Java
341   // NativeInterpreterWrapper instance.
342   return reinterpret_cast<jlong>(interpreter.release());
343 }

346 JNIEXPORT void JNICALL Java_org_tensorflow_lite_NativeInterpreterWrapper_run(
347     JNIEnv* env, jclass clazz, jlong interpreter_handle, jlong error_handle) {
348   tflite::Interpreter* interpreter =
349       convertLongToInterpreter(env, interpreter_handle);
350   if (interpreter == nullptr) return;
351   BufferErrorReporter* error_reporter =
352       convertLongToErrorReporter(env, error_handle);
353   if (error_reporter == nullptr) return;
354 
355   if (interpreter->Invoke() != kTfLiteOk) {
356     throwException(env, kIllegalArgumentException,
357                    "Internal error: Failed to run on the given Interpreter: %s",
358                    error_reporter->CachedErrorMessage());
359     return;
360   }
361 }

363 JNIEXPORT jint JNICALL
364 Java_org_tensorflow_lite_NativeInterpreterWrapper_getOutputDataType(
365     JNIEnv* env, jclass clazz, jlong handle, jint output_idx) {
366   tflite::Interpreter* interpreter = convertLongToInterpreter(env, handle);
367   if (interpreter == nullptr) return -1;
368   const int idx = static_cast<int>(output_idx);
369   if (output_idx < 0 || output_idx >= interpreter->outputs().size()) {
370     throwException(env, kIllegalArgumentException,
371                    "Failed to get %d-th output out of %d outputs", output_idx,
372                    interpreter->outputs().size());
373     return -1;
374   }
375   TfLiteTensor* target = interpreter->tensor(interpreter->outputs()[idx]);
376   int type = getDataType(target->type);
377   return static_cast<jint>(type);
378 }

412 JNIEXPORT jboolean JNICALL
413 Java_org_tensorflow_lite_NativeInterpreterWrapper_resizeInput(
414     JNIEnv* env, jclass clazz, jlong interpreter_handle, jlong error_handle,
415     jint input_idx, jintArray dims) {
416   BufferErrorReporter* error_reporter =
417       convertLongToErrorReporter(env, error_handle);
418   if (error_reporter == nullptr) return JNI_FALSE;
419   tflite::Interpreter* interpreter =
420       convertLongToInterpreter(env, interpreter_handle);
421   if (interpreter == nullptr) return JNI_FALSE;
422   const int idx = static_cast<int>(input_idx);
423   if (idx < 0 || idx >= interpreter->inputs().size()) {
424     throwException(env, kIllegalArgumentException,
425                    "Input error: Can not resize %d-th input for a model having "
426                    "%d inputs.",
427                    idx, interpreter->inputs().size());
428     return JNI_FALSE;
429   }
430   // check whether it is resizing with the same dimensions.
431   TfLiteTensor* target = interpreter->tensor(input_idx);
432   bool is_changed = areDimsDifferent(env, target, dims);
433   if (is_changed) {
434     TfLiteStatus status = interpreter->ResizeInputTensor(
435         interpreter->inputs()[idx], convertJIntArrayToVector(env, dims));
436     if (status != kTfLiteOk) {
437       throwException(env, kIllegalArgumentException,
438                      "Internal error: Failed to resize %d-th input: %s", idx,
439                      error_reporter->CachedErrorMessage());
440       return JNI_FALSE;
441     }
442   }
443   return is_changed ? JNI_TRUE : JNI_FALSE;
444 }

446 JNIEXPORT void JNICALL
447 Java_org_tensorflow_lite_NativeInterpreterWrapper_applyDelegate(
448     JNIEnv* env, jclass clazz, jlong interpreter_handle, jlong error_handle,
449     jlong delegate_handle) {
450   tflite::Interpreter* interpreter =
451       convertLongToInterpreter(env, interpreter_handle);
452   if (interpreter == nullptr) return;
453 
454   BufferErrorReporter* error_reporter =
455       convertLongToErrorReporter(env, error_handle);
456   if (error_reporter == nullptr) return;
457 
458   TfLiteDelegate* delegate = convertLongToDelegate(env, delegate_handle);
459   if (delegate == nullptr) return;
460 
461   TfLiteStatus status = interpreter->ModifyGraphWithDelegate(delegate);
462   if (status != kTfLiteOk) {
463     throwException(env, kIllegalArgumentException,
464                    "Internal error: Failed to apply delegate: %s",
465                    error_reporter->CachedErrorMessage());
466   }
467 }
